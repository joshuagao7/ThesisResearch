% Slide 5: Human Intuition
\begin{frame}{Human Intuition for Analyzing Jumps}
Vertical jumps exhibit a characteristic signature:
\begin{enumerate}
    \item Force build-up during countermovement and propulsion
    \item Rapid unloading at takeoff
    \item Near-zero force while airborne
    \item Rapid reloading at landing
\end{enumerate}
\vspace{0.3cm}
\textbf{Challenge:} While patterns are visually obvious to humans, automated detection in noisy real-world conditions is difficult.
\end{frame}

% Slide 10: Threshold and Derivative Algorithms
\begin{frame}{Threshold \& Derivative Algorithms}
\textbf{Threshold Algorithm:}
\begin{itemize}
    \item Identifies flight periods when force falls below threshold $\theta$
    \item Binary mask: $m_{\text{threshold}}(t) = 1$ if $\bar{d}(t) < \theta$
    \item Filtered by physics constraints: $t_{\min} \approx 0.2$~s, $t_{\max} \approx 1.2$~s
\end{itemize}
\vspace{0.5cm}
\textbf{Derivative Algorithm:}
\begin{itemize}
    \item Detects paired takeoff and landing events in derivative signal
    \item Upper threshold: $d'(t) > \theta_{\text{upper}}$ (landing)
    \item Lower threshold: $d'(t) < \theta_{\text{lower}}$ (takeoff)
    \item Validated by in-air threshold: $\bar{d}(t) < \theta_{\text{air}}$ during flight
\end{itemize}
\vspace{0.3cm}
Both algorithms operate in $O(T)$ time, enabling real-time mobile deployment.
\end{frame}

% Slide 12: Scoring and Loss Function
\begin{frame}{Scoring Each Algorithm: Loss Function}
\textbf{Loss Function:}
\begin{equation*}
L(\theta) = \text{FP}(\theta) + \text{FN}(\theta)
\end{equation*}
\vspace{0.3cm}
\begin{itemize}
    \item \textbf{False Positives (FP):} Detected jumps without corresponding ground truth markers
    \item \textbf{False Negatives (FN):} Ground truth markers not captured by any detected jump
\end{itemize}
\vspace{0.5cm}
\textbf{Manual Tagging:} All jump events manually annotated in dataset
\end{frame}

% Slide 14: Langevin Monte Carlo Sampling Methodology
\begin{frame}{Langevin Monte Carlo Sampling Optimization}
\textbf{Methodology:} Stochastic optimization technique inspired by Langevin dynamics for efficient exploration of high-dimensional parameter spaces.
\vspace{0.3cm}
\begin{itemize}
    \item \textbf{Metropolis-Hastings acceptance criterion:} Accepts parameter changes that decrease loss, or probabilistically accepts increases based on temperature
    \item \textbf{Temperature schedule:} Initial temperature 10.0 with exponential decay rate 0.9995 per iteration
    \item \textbf{Escape from local minima:} Temperature schedule enables exploration beyond immediate improvements
    \item \textbf{Comprehensive optimization:} 10,000 iterations for multi-parameter optimization across all algorithms
\end{itemize}
\vspace{0.2cm}
\textbf{Advantages over grid search:}
\begin{itemize}
    \item Computationally feasible for high-dimensional spaces (e.g., 5+ parameters)
    \item Efficiently explores parameter space without exhaustive enumeration
    \item Finds optimal configurations that minimize loss function $L(\theta) = \text{FP}(\theta) + \text{FN}(\theta)$
\end{itemize}
\end{frame}
